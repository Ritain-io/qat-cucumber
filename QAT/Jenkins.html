def self.register_vars(opts={})
  ignore_list = opts[:ignore] || []
  add_list    = opts[:add] || []

  list = DEFAULT_ENV_VARS + add_list - ignore_list

  list.each do |var|
    Log4r::MDC.put var, ENV[var] if ENV[var]
  end

end

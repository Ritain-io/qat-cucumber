def execute!
  parse_argv!

  handle_options if @options.any?

rescue ArgumentError, OptionParser::MissingArgument => exception
  @stderr.puts "Error: #{exception.message}"
  @exit_code = 1
ensure
  @kernel.exit(exit_code)
end
